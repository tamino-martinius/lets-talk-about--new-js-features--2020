Intl.DateTimeFormat

const DateFormat = Intl.DateTimeFormat;
new DateFormat('en-US').format(date);
// ‚Üí '12/20/2019'
new DateFormat('en-GB').format(date);
// ‚Üí '20/12/2019'
// Include a fallback language
new DateFormat(['foo', 'de']).format(date)
// ‚Üí '20.12.2019'

Intl.NumberFormat

new Intl.NumberFormat('de-DE', {
  style: 'currency', currency: 'EUR'
}).format(number); // ‚Üí '123.456,79 ‚Ç¨'
// the Japanese yen doesn't use a minor unit
new Intl.NumberFormat('ja-JP', {
  style: 'currency', currency: 'JPY'
}).format(number); // ‚Üí 'Ôø•123,457'
new Intl.NumberFormat('en-IN', {
  maximumSignificantDigits: 3
}).format(number); // ‚Üí '1,23,000'

Intl.PluralRules

new Intl.PluralRules('en-US').select(0);
// ‚Üí 'other'
new Intl.PluralRules('en-US').select(1);
// ‚Üí 'one'
new Intl.PluralRules('en-US').select(2);
// ‚Üí 'other'

Intl.Collator

function letterSort(lang, letters) {
  letters.sort(new Intl.Collator(lang).compare);
  return letters;
}
console.log(letterSort('de', ['a','z','√§']));
// expected output: Array ["a", "√§", "z"]
console.log(letterSort('sv', ['a','z','√§']));
// expected output: Array ["a", "z", "√§"]
